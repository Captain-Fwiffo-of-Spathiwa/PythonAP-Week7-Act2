def binary_search(tasks, target):
    # pdb.set_trace()
    sort_tasks(tasks)
    low, high = 0, len(tasks) - 1
    while low <= high:
        mid = (low + high)
        if tasks[mid][0] == target:
            return mid
        elif tasks[mid][0] < target:
            low = mid + 1
        else:
            high = mid - 1
    return -1
